<div class="game-box">

  <h1> 血管保衛戰</h1>

  <p id="question"></p>

  <div id="answers"></div>

  <p id="score">血管健康值：100</p>

</div>


<script>

  let score = 100;

  let current = 0;


  const quiz = [

    {

      q: "哪一種膽固醇最容易造成血管粥狀硬化？",

      a: ["HDL-C", "LDL-C"],

      correct: 1

    },

    {

      q: "哪一種膽固醇可以幫忙清除多餘膽固醇？",

      a: ["LDL-C", "HDL-C"],

      correct: 1

    },

    {

      q: "以下哪一種食物有助於降低血脂？",

      a: ["甜甜圈", "深海魚（Omega-3）"],

      correct: 1

    },

    {

      q: "哪一個行為有助於預防高血脂？",

      a: ["每天適度運動", "久坐不動"],

      correct: 0

    },

    {

      q: "吸菸對血管的影響是？",

      a: ["破壞血管，增加膽固醇堆積", "對血管沒有影響"],

      correct: 0

    },

    {

      q: "一般年輕人建議多久檢查一次血脂？",

      a: ["每3～5年", "每天"],

      correct: 0

    }

  ];


  function showQuestion() {

    if (current >= quiz.length) {

      endGame();

      return;

    }


    document.getElementById("question").innerText =

      "第 " + (current + 1) + " 題：" + quiz[current].q;


    const answersDiv = document.getElementById("answers");

    answersDiv.innerHTML = "";


    quiz[current].a.forEach((text, index) => {

      const btn = document.createElement("button");

      btn.innerText = text;

      btn.onclick = () => checkAnswer(index);

      answersDiv.appendChild(btn);

    });

  }


  function checkAnswer(choice) {

    if (choice === quiz[current].correct) {

      alert("答對了！血管更健康 ");

    } else {

      alert("答錯了！血管受損 ");

      score -= 10;

    }


    document.getElementById("score").innerText =

      "血管健康值：" + score;


    current++;

    showQuestion();

  }


  function endGame() {

    let result = "遊戲結束！\n\n";

    if (score >= 80) {

      result += "血管超健康守護者！";

    } else if (score >= 60) {

      result += " 血管小達人，繼續加油！";

    } else {

      result += "壞膽固醇來襲，要更注意飲食與生活習慣！";

    }

    document.querySelector(".game-box").innerHTML =

      "<h2>" + result + "</h2><p>記得：少油少糖、多運動、戒菸、定期檢查 </p>";

  }


  showQuestion();

</script>
